// --- FUNÇÃO SETUP ---
void setup() {
    Serial.begin(115200);
    delay(2000);
    Serial.println("Robô PID com Guia Central + Bigodes");

    // Configura os pinos dos sensores
    pinMode(PINO_SENSOR_CENTRAL, INPUT);
    pinMode(PINO_BIGODE_ESQ_DENTRO, INPUT);
    pinMode(PINO_BIGODE_ESQ_FORA, INPUT);
    pinMode(PINO_BIGODE_DIR_DENTRO, INPUT);
    pinMode(PINO_BIGODE_DIR_FORA, INPUT);

    pinMode(PINO_LED, OUTPUT);
    digitalWrite(PINO_LED, LOW);

    // Configura o Wi-Fi e Servidor Web
    WiFi.mode(WIFI_AP);
    WiFi.softAP("Vespa PID", "12345678");
    configurar_servidor_web();
    server.begin();
    Serial.print("Servidor iniciado em: ");
    Serial.println(WiFi.softAPIP());

    // Carrega valores PID salvos
    SPIFFS.begin(DIRETORIO_SPIFFS, false);
    espera = SPIFFS.getFloat(ALIAS_ESPERA, 10.0);
    Kp = SPIFFS.getFloat(ALIAS_KP, 10.0);
    Ki = SPIFFS.getFloat(ALIAS_KI, 0.1);
    Kd = SPIFFS.getFloat(ALIAS_KD, 20.0);
    SPIFFS.end();
}


// --- FUNÇÃO LOOP ---
void loop() {
    ws.cleanupClients();
    if (!parada) {
        segue_linha();
    }
}

// ------------------------------------------------------------------
// <<<<<<<<<<<<<<<<<<<< O CORAÇÃO DO CÓDIGO >>>>>>>>>>>>>>>>>>>>>>>
// ------------------------------------------------------------------
void segue_linha() {
    // Lê os "bigodes" digitais. LOW (0) significa que o sensor VÊ a linha.
    bool bigode_esq_fora   = (digitalRead(PINO_BIGODE_ESQ_FORA)   == LOW);
    bool bigode_esq_dentro = (digitalRead(PINO_BIGODE_ESQ_DENTRO) == LOW);
    bool bigode_dir_dentro = (digitalRead(PINO_BIGODE_DIR_DENTRO) == LOW);
    bool bigode_dir_fora   = (digitalRead(PINO_BIGODE_DIR_FORA)   == LOW);
    
    // Lê o sensor analógico central (Lógica invertida na versão anterior)
    bool central_na_linha = (analogRead(PINO_SENSOR_CENTRAL) < LEITURA_LINHA_CENTRAL);

    // --- LÓGICA HÍBRIDA PARA CALCULAR O ERRO NUMÉRICO ---
    // Esta sequência de 'if/else' garante que a condição mais externa (mais crítica)
    // seja processada primeiro, definindo um valor de erro para o PID.
    
    if (bigode_esq_fora) {
        erro = -4; // Erro máximo para a esquerda
    } else if (bigode_dir_fora) {
        erro = 4;  // Erro máximo para a direita
    } else if (bigode_esq_dentro) {
        erro = -2; // Erro médio para a esquerda
    } else if (bigode_dir_dentro) {
        erro = 2;  // Erro médio para a direita
    } else if (central_na_linha) {
        erro = 0;  // Perfeitamente na linha
    } else {
        // CONDIÇÃO DE LINHA PERDIDA:
        // Quando nenhum sensor vê a linha, o robô segue a lógica do 
        // sensor central (como se estivesse na linha), resultando em erro = 0.
        // Isso fará o robô seguir em frente, útil para passar por falhas na pista.
        erro = 0;
    }

    // Se encontrou a linha de alguma forma, reseta o contador de "perdido"
    // Com a nova lógica, os valores 5 e -5 não são mais usados para "perdido".
    if (erro != 5 && erro != -5) { 
        contador_perdido = 0;
    } else {
        contador_perdido++;
    }

    // Define a velocidade atual baseada na situação
    int velocidade_atual;
    if (contador_perdido > 0) {
        velocidade_atual = velocidade_base / 2; // Reduz a velocidade para procurar a linha
    } else if (erro == 0) {
        velocidade_atual = velocidade_base + VELOCIDADE_RETA_ACRESCIMO; // Acelera em retas
    } else {
        velocidade_atual = velocidade_base; // Velocidade normal em curvas
    }
    
    // Se ficou perdido por muito tempo, para
    if (contador_perdido > CONTAGEM_MAXIMA_PERDIDO) {
        motores.stop();
        I = 0; // Reseta o termo integral
    } else {
        // Calcula a resposta do PID e aciona os motores
        calcula_PID(velocidade_atual);
        motores.turn(velocidade_direita, velocidade_esquerda);
    }

    delay((uint32_t)espera);
}


// ------------------------------------------------------------------
// <<<<<<<<<<<< FUNÇÃO PID CORRIGIDA >>>>>>>>>>>>>>
// ------------------------------------------------------------------
void calcula_PID(int velocidade_atual) {
    P = erro;
    I = I + erro;

    // MUDANÇA 1: Limpa a "memória" (Integral) quando o robô está reto.
    // Isso evita que erros acumulados de curvas passadas atrapalhem a reta.
    if (erro == 0) {
      I = 0;
    }
    // Limita o valor de I para evitar que ele cresça demais.
    I = constrain(I, -300, 300); 

    D = erro - erro_anterior;
    resposta_PID = (Kp * P) + (Ki * I) + (Kd * D);
    erro_anterior = erro;

    // MUDANÇA 2: Sinais da correção invertidos para o lado certo.
    // Agora a resposta do PID corrige a direção corretamente.
    velocidade_direita = velocidade_atual + resposta_PID;
    velocidade_esquerda = velocidade_atual - resposta_PID;

    // Limita a velocidade dos motores para não exceder o máximo
    velocidade_direita = constrain(velocidade_direita, -VELOCIDADE_MAXIMA, VELOCIDADE_MAXIMA);
    velocidade_esquerda = constrain(velocidade_esquerda, -VELOCIDADE_MAXIMA, VELOCIDADE_MAXIMA);
}


// --- FUNÇÕES DO SERVIDOR WEB E WEBSOCKET ---
void configurar_servidor_web(void) {
    ws.onEvent(onEvent);
    server.addHandler(&ws);
    server.on("/", HTTP_GET, [](AsyncWebServerRequest * request) {
        request->send_P(200, "text/html", index_html);
    });
}

void handleWebSocketMessage(void *arg, uint8_t *data, size_t len) {
    AwsFrameInfo *info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
        data[len] = 0;
        StaticJsonDocument<256> json;
        DeserializationError error = deserializeJson(json, (char*)data);
        if (error) return;

        if (json.containsKey(ALIAS_PARA)) {
            parada = true;
            I = 0; // Reseta o termo integral
            motores.stop();
        } else if (json.containsKey(ALIAS_KP)) {
            parada = false;
            espera = json[ALIAS_ESPERA];
            Kp = json[ALIAS_KP];
            Ki = json[ALIAS_KI];
            Kd = json[ALIAS_KD];
            // Salva os novos valores na memória
            SPIFFS.begin(DIRETORIO_SPIFFS, false);
            SPIFFS.putFloat(ALIAS_ESPERA, espera);
            SPIFFS.putFloat(ALIAS_KP, Kp);
            SPIFFS.putFloat(ALIAS_KI, Ki);
            SPIFFS.putFloat(ALIAS_KD, Kd);
            SPIFFS.end();
        } else if (json.containsKey(ALIAS_VELOCIDADE)) {
            String dir = json[ALIAS_VELOCIDADE];
            if (dir == "up") velocidade_base += 5;
            else if (dir == "down") velocidade_base -= 5;
            velocidade_base = constrain(velocidade_base, VELOCIDADE_MINIMA, VELOCIDADE_MAXIMA);
            // Envia nova velocidade para todos os clientes
            StaticJsonDocument<JSON_OBJECT_SIZE(1)> speed_json;
            speed_json[ALIAS_VELOCIDADE] = velocidade_base;
            String mensagem;
            serializeJson(speed_json, mensagem);
            ws.textAll(mensagem);
        }
    }
}

void onEvent(AsyncWebSocket *server, AsyncWebSocketClient *client, AwsEventType type, void *arg, uint8_t *data, size_t len) {
    switch (type) {
        case WS_EVT_CONNECT: {
            digitalWrite(PINO_LED, HIGH);
            StaticJsonDocument<JSON_OBJECT_SIZE(5)> json;
            json[ALIAS_KP] = Kp;
            json[ALIAS_KI] = Ki;
            json[ALIAS_KD] = Kd;
            json[ALIAS_ESPERA] = espera;
            json[ALIAS_VELOCIDADE] = velocidade_base;
            String mensagem;
            serializeJson(json, mensagem);
            client->text(mensagem);
            break;
        }
        case WS_EVT_DISCONNECT:
            if (ws.count() == 0) digitalWrite(PINO_LED, LOW);
            break;
        case WS_EVT_DATA:
            handleWebSocketMessage(arg, data, len);
            break;
        default:
            break;
    }
}
